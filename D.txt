#!/bin/bash

read n;

array=();

for ((i = 0; i < n; i++)) do
    read item;
    array+=($item);
done


max=-10000000000;
indexMax=-1;
min=10000000000;
indexMin=-1;

for ((i = 0; i < n; i++)) do
    if [[ ${array[i]} -gt ${max} ]]
    then
        ((max=array[i]));
        ((indexMax=i));
    else
        if [[ ${array[i]} -lt ${min} ]]
        then
            min=$array[i];
            ((indexMin=i));
        fi
    fi
done

for ((i = 0; i < n; i++)) do
    if [[ i=$indexMax ]]
    then
      echo ${max};
    else
      if [[ i=$indexMin ]];
      then
        echo ${min};
      else
        echo ${array[i]};
      fi;
done
